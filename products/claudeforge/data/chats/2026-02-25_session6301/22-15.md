cat > ~/Empire/scripts/autosave-chat.sh << 'EOF'
#!/bin/bash
# Auto-save latest Claude chat summary every 60 seconds
ARCHIVE_DIR="$HOME/Empire/docs/CHAT_ARCHIVE"
LATEST="$ARCHIVE_DIR/latest_claude.md"
mkdir -p "$ARCHIVE_DIR"

while true; do
  TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')
  # Touch the latest file with timestamp
  if [ -f "$LATEST" ]; then
    sed -i "1s/.*/# Last auto-save: $TIMESTAMP/" "$LATEST" 2>/dev/null
  fi
  sleep 60
done
EOF
chmod +x ~/Empire/scripts/autosave-chat.sh
